```bash
# 1. Install Firebase CLI (once)
npm install -g firebase-tools

# 2. Login to Firebase (once)
firebase login

# 3. Initialize Firebase in your project
firebase init

# 4. Build your React project (for deployment)
npm run build

# 5. Deploy to Firebase Hosting
firebase deploy

# 6. Serve locally (to test Firebase config)
firebase serve

# 7. Check current Firebase project config
firebase projects:list

# 8. Logout (optional)
firebase logout
```
